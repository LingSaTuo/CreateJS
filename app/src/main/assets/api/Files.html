非映射API (Files)

该类位于com.lingsatuo.openapi.Files

该类集成自libs_utils

copy(fromString,toString)

复制文件 把位于fromString的文件复制到toString,错误时抛出异常

var obj = new com.lingsatuo.openapi.Files()
obj.copy("/sdcard/1.js","/sdcard/a")

move(fromString,toString)

移动文件 把位于fromString的文件移动到toString,错误时抛出异常

var obj = new com.lingsatuo.openapi.Files()
obj.move("/sdcard/1.js","/sdcard/a")

rename(pathString,nameString)

把位于pathString的文件重名为nameString,返回是否成功（true,false）

var obj = new com.lingsatuo.openapi.Files()
obj.rename("/sdcard/1.js","/sdcard/2.js")

getNameWithOutExtension(path)

获取位于path的文件的名字，并去除拓展名

var obj = new com.lingsatuo.openapi.Files()
obj.getNameWithOutExtension("/sdcard/app.js")//返回app

remove(path)

移除位于path的文件或者空文件夹，返回是否成功

var obj = new com.lingsatuo.openapi.Files()
obj.remove("/sdcard/1.js")

removeDir(path)

删除位于path的文件或者文件夹 返回是否成功

var obj = new com.lingsatuo.openapi.Files()
obj.removeDir("/sdcard/1.js")

getSdcardPath()

返回SD卡路径

getExtension(nameString)

返回nameString的后缀

var obj = new com.lingsatuo.openapi.Files()
obj.removeDir("1.js")//返回js

listDir(path[,filter])

返回path下的所有文件 +filter过滤条件 接受一个String类型的参数，返回boolean值，可为空

var obj = new com.lingsatuo.openapi.Files()
obj.listDir("/sdcard/",function(name){
    return name.endsWith(".js")|| name.endsWith(".txt")
}) //这样就可以返回path下所有的js文件（夹）和txt文件（夹）

readFile(pathString,encoding)

用encoding编码方式读取位于pathString的文件，并返回文件内容

var obj = new com.lingsatuo.openapi.Files()
obj.readFile("/sdcard/1.js","utf-8")

getMimeType(pathString)

获取文件的mimetype

var obj = new com.lingsatuo.openapi.Files()
obj.getMimeType("/sdcard/1.js")

file2Base64(pathString,toString)

把位于pathString的文件转成Base64编码，并保存在toString路径，返回值为Base64，toString 可为null,但不能不写

var obj = new com.lingsatuo.openapi.Files()
obj.file2Base64("/sdcard/1.js","/sdcard/1.txt")

base642File(encodeString,toString)

把encodeString字符转为文件，并保存在toString路径

var obj = new com.lingsatuo.openapi.Files()
obj.base642File("此处为编码，不是路径","/sdcard/1.js")
